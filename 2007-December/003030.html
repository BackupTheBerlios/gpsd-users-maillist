Here's my ugly little config generator.

#!/bin/sh

make -i distclean
if [ $# -eq 1 -a x"$1" = x"-r" ] ; then
	export AUTOCONF_VERSION=2.60 AUTOMAKE_VERSION=1.9
	rm -f missing
	aclocal
	libtoolize --force --copy
	autoheader --force
	automake --add-missing --foreign --copy --include-deps
	autoconf --force
	unset AUTOCONF_VERSION AUTOMAKE_VERSION
fi

env	FOO="bar" \
	X_LIBS="-lXm -lXaw -lXmu -lXt -lICE -lX11" \
	CPPFLAGS="-I/usr/local/include -I/usr/X11R6/include" \
	LDFLAGS="-L/usr/X11R6/lib -L/usr/local/lib -g" \
	X_LIBS="" CPPFLAGS="" LDFLAGS="-g -static" \
	CFLAGS="-g -O0 -W -Wall -Wno-format-y2k -static" \
	libtool_flags="-static -all-static" \
	CC="cc" \
./configure --prefix=/home/gpsd --disable-shared \
	--without-x --enable-python \
	--disable-pps --disable-ntpshm \
	--enable-gpsd-user=_gpsd \
	--enable-reconfigure \
	--enable-sirf \
	--enable-nmea \
	--enable-tsip \
	--enable-tripmate \
	--enable-earthmate \
	--enable-evermore \
	--enable-garmin \
	--enable-ubx \
	--enable-navcom \
	--enable-itrax \
	--disable-tnt \
	--disable-fv18 \
	--disable-raw \
	--enable-rtcm104 --disable-ntrip


-- 
GDB has a 'break' feature; why doesn't it have 'fix' too?
