/* gpslogger.c - Tilman Koschnick <til at subnetz.org> - January 2005 */

#include <math.h>
#include <stdlib.h>
#include <stdio.h>
#include <time.h>
#include <unistd.h>
#include <gps.h>

static struct gps_data_t *gpsdata;

int main (int argc, char *argv[])
{
   if (!(gpsdata = gps_open(NULL, NULL))) {
      fputs("gpslogger: no gpsd running or network error\n", stderr);
      exit(2);
   }

   gps_query(gpsdata, "w\n");
   for (;;) {
      (void) gps_poll(gpsdata);
      if ( gpsdata->status > 0 ) {
         printf("FIX\n");
      } else {
         printf("NO FIX\n");
      }
   }

   return 0;
}
